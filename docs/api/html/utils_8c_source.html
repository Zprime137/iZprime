18static char *dup_trimmed(const char *src)
19{
20 if (src == NULL)
21 return NULL;
22
23 const char *start = skip_spaces(src);
24 const char *end = start + strlen(start);
25 while (end > start && isspace((unsigned char)end[-1]))
26 --end;
27
28 size_t len = (size_t)(end - start);
29 char *out = malloc(len + 1);
30 if (!out)
31 return NULL;
32 memcpy(out, start, len);
33 out[len] = '\0';
34 return out;

37static char *normalize_decimal_token(const char *token)
38{
39 char *trimmed = dup_trimmed(token);
40 if (trimmed == NULL)
41 return NULL;
42
43 const char *s = trimmed;
44 if (*s == '+')
45 ++s;
46
47 if (*s == '\0')
48 {
49 free(trimmed);
50 return NULL;
51 }
52
53 size_t cap = strlen(s) + 1;
54 char *normalized = malloc(cap);
55 if (!normalized)
56 {
57 free(trimmed);
58 return NULL;
59 }
60
61 size_t n = 0;
62 int has_comma = strchr(s, ',') != NULL;
63
64 if (!has_comma)
65 {
66 for (const char *p = s; *p; ++p)
67 {
68 if (*p == '_')
69 continue;
70 if (!isdigit((unsigned char)*p))
71 {
72 free(normalized);
73 free(trimmed);
74 return NULL;
75 }
76 normalized[n++] = *p;
77 }
78 }
79 else
80 {
81 const char *seg = s;
82 int group_idx = 0;
83 while (1)
84 {
85 const char *comma = strchr(seg, ',');
86 size_t seg_len = comma ? (size_t)(comma - seg) : strlen(seg);
87 if (seg_len == 0)
88 {
89 free(normalized);
90 free(trimmed);
91 return NULL;
92 }
93
94 if (group_idx == 0)
95 {
96 if (seg_len < 1 || seg_len > 3)
97 {
98 free(normalized);
99 free(trimmed);
100 return NULL;
101 }
102 }
103 else if (seg_len != 3)
104 {
105 free(normalized);
106 free(trimmed);
107 return NULL;
108 }
109
110 for (size_t i = 0; i < seg_len; ++i)
111 {
112 if (!isdigit((unsigned char)seg[i]))
113 {
114 free(normalized);
115 free(trimmed);
116 return NULL;
117 }
118 normalized[n++] = seg[i];
119 }
120
121 group_idx++;
122 if (comma == NULL)
123 break;
124 seg = comma + 1;
125 }
126 }
127
128 normalized[n] = '\0';
129 free(trimmed);
130 return normalized;
131}

348 if (len > 7 && strncmp(range, "range[", 6) == 0 && range[len - 1] == ']')

359 }

362 if (sep != NULL)

371 if (sep != NULL)

373 *sep = '\0';
374 int ok = parse_range_parts(range, sep + 1, lower, upper);
375 free(range);

377 }

453uint64_t modular_inverse(uint64_t a, uint64_t m)
454{
455 if (m == 1)
456 return 0;
457
458 uint64_t m0 = m;
459 int64_t x0 = 0, x1 = 1;
460
461 while (a > 1)
462 {
463 uint64_t q = a / m;
464 uint64_t t = m;
465
466 m = a % m;
467 a = t;
468 t = x0;
469
470 x0 = x1 - (int64_t)q * x0;
471 x1 = t;
472 }
473
474 if (x1 < 0)
475 x1 += (int64_t)m0;
476 return (uint64_t)x1;
477}

531int get_cpu_L2_cache_size_bits(void)
532{
533
534#ifdef __linux__
535 // Linux: Try reading from sysfs
536 FILE *fp = fopen("/sys/devices/system/cpu/cpu0/cache/index2/size", "r");
537 if (fp != NULL)
538 {
539 char buffer[32];
540 if (fgets(buffer, sizeof(buffer), fp) != NULL)
541 {
542 // Parse size with K suffix (e.g., "256K")
543 int size_kb = 0;
544 if (sscanf(buffer, "%dK", &size_kb) == 1)
545 {
546 fclose(fp);
547 return size_kb * 1024 * 8; // Convert KB to bits
548 }
549 }
550 fclose(fp);
551 }
552#endif
553
554#if defined(__APPLE__) || defined(__FreeBSD__) || defined(__OpenBSD__) || defined(__NetBSD__)
555 // macOS and BSD: Use sysctl
556 size_t size_bytes = 0;
557 size_t len = sizeof(size_bytes);
558
559 // Try hw.l2cachesize (macOS/some BSDs)
560 if (sysctlbyname("hw.l2cachesize", &size_bytes, &len, NULL, 0) == 0)
561 {
562 if (size_bytes > 0)
563 {
564 return (int)(size_bytes * 8); // Convert bytes to bits
565 }
566 }
567
568 // Fallback: Try machdep.cpu.cache.L2_cache_size (older macOS)
569 if (sysctlbyname("machdep.cpu.cache.L2_cache_size", &size_bytes, &len, NULL, 0) == 0)
570 {
571 if (size_bytes > 0)
572 {
573 return (int)(size_bytes * 8); // Convert bytes to bits
574 }
575 }
576#endif
577
578 // Conservative fallback: 256 KB is common for L2 cache per core
579 return 256 * 1024 * 8; // Convert 256 KB to bits
580}
