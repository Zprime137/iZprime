57BITMAP *bitmap_init(size_t size, int set_bits)
58{
59 assert(size > 0 && "Bitmap size must be positive");
60
61 BITMAP *bitmap = (BITMAP *)malloc(sizeof(BITMAP));
62 if (bitmap == NULL)
63 {
64 log_error("Memory allocation failed for BITMAP struct");
65 return NULL;
66 }
67
68 bitmap->size = size;
69 bitmap->byte_size = (size + 7) / 8;
70 bitmap->data = (unsigned char *)malloc(bitmap->byte_size);
71 if (bitmap->data == NULL)
72 {
73 free(bitmap);
74 log_error("bitmap_create: Memory allocation failed for BITMAP");
75 return NULL;
76 }
77
78 if (set_bits)
79 memset(bitmap->data, 0xFF, bitmap->byte_size);
80 else
81 memset(bitmap->data, 0x00, bitmap->byte_size);
82
83 memset(bitmap->sha256, 0, SHA256_DIGEST_LENGTH); // Initialize SHA-256 to zero
84
85 return bitmap;
86}

233void bitmap_clear_steps_simd(BITMAP *bitmap, uint64_t step, uint64_t start_idx, uint64_t limit)
234{
235 assert(step > 0 && "Step must be positive in bitmap_clear_steps_simd.");
236 limit = MIN(limit, bitmap->size - 1);
237
238#ifdef __aarch64__
239 // NEON implementation for ARM64
240 uint64_t idx = start_idx;
241
242 // Process 4 steps at a time
243 // Condition: idx + 3*step <= limit
244
245 if (limit >= 3 * step && idx <= limit - 3 * step)
246 {
247 uint64x2_t v_step2 = vdupq_n_u64(2 * step);
248 uint64x2_t v_step4 = vdupq_n_u64(4 * step);
249
250 // Initialize indices: [idx, idx+step] and [idx+2step, idx+3step]
251 uint64x2_t v_idx_01 = vdupq_n_u64(idx);
252 v_idx_01 = vsetq_lane_u64(idx + step, v_idx_01, 1);
253
254 uint64x2_t v_idx_23 = vaddq_u64(v_idx_01, v_step2);
255
256 while (idx <= limit - 3 * step)
257 {
258 // Extract indices and perform operations
259 // Note: NEON doesn't have efficient scatter for bytes, so we extract and do scalar stores
260
261 uint64_t i0 = vgetq_lane_u64(v_idx_01, 0);
262 uint64_t i1 = vgetq_lane_u64(v_idx_01, 1);
263 uint64_t i2 = vgetq_lane_u64(v_idx_23, 0);
264 uint64_t i3 = vgetq_lane_u64(v_idx_23, 1);
265
266 bitmap->data[i0 / 8] &= ~(1 << (i0 % 8));
267 bitmap->data[i1 / 8] &= ~(1 << (i1 % 8));
268 bitmap->data[i2 / 8] &= ~(1 << (i2 % 8));
269 bitmap->data[i3 / 8] &= ~(1 << (i3 % 8));
270
271 // Advance indices
272 v_idx_01 = vaddq_u64(v_idx_01, v_step4);
273 v_idx_23 = vaddq_u64(v_idx_23, v_step4);
274
275 idx += 4 * step;
276 }
277 }
278
279 // Handle remaining steps
280 for (; idx <= limit; idx += step)
281 {
282 bitmap->data[idx / 8] &= ~(1 << (idx % 8));
283 }
284
285#elif defined(__AVX2__)
286 // AVX2 implementation for x86_64
287 uint64_t idx = start_idx;
288
289 // Process 4 steps at a time
290 if (limit >= 3 * step && idx <= limit - 3 * step)
291 {
292 __m256i v_step4 = _mm256_set1_epi64x(4 * step);
293
294 // Initialize indices: [idx, idx+step, idx+2step, idx+3step]
295 // Note: _mm256_set_epi64x takes arguments in reverse order (e3, e2, e1, e0)
296 __m256i v_idx = _mm256_set_epi64x(idx + 3 * step, idx + 2 * step, idx + step, idx);
297
298 while (idx <= limit - 3 * step)
299 {
300 // Extract indices
301 uint64_t i0 = _mm256_extract_epi64(v_idx, 0);
302 uint64_t i1 = _mm256_extract_epi64(v_idx, 1);
303 uint64_t i2 = _mm256_extract_epi64(v_idx, 2);
304 uint64_t i3 = _mm256_extract_epi64(v_idx, 3);
305
306 bitmap->data[i0 / 8] &= ~(1 << (i0 % 8));
307 bitmap->data[i1 / 8] &= ~(1 << (i1 % 8));
308 bitmap->data[i2 / 8] &= ~(1 << (i2 % 8));
309 bitmap->data[i3 / 8] &= ~(1 << (i3 % 8));
310
311 // Advance indices
312 v_idx = _mm256_add_epi64(v_idx, v_step4);
313 idx += 4 * step;
314 }
315 }
316
317 // Handle remaining steps
318 for (; idx <= limit; idx += step)
319 {
320 bitmap->data[idx / 8] &= ~(1 << (idx % 8));
321 }
322
323#elif defined(__SSE2__)
324 // SSE2 implementation for x86_64
325 uint64_t idx = start_idx;
326
327 // Process 2 steps at a time
328 if (limit >= step && idx <= limit - step)
329 {
330 __m128i v_step2 = _mm_set1_epi64x(2 * step);
331 // Note: _mm_set_epi64x takes arguments in reverse order (e1, e0)
332 __m128i v_idx = _mm_set_epi64x(idx + step, idx);
333
334 while (idx <= limit - step)
335 {
336 // Extract indices
337 // _mm_cvtsi128_si64 extracts the lower 64 bits (element 0)
338 uint64_t i0 = _mm_cvtsi128_si64(v_idx);
339 // Unpack high to low to extract element 1
340 uint64_t i1 = _mm_cvtsi128_si64(_mm_unpackhi_epi64(v_idx, v_idx));
341
342 bitmap->data[i0 / 8] &= ~(1 << (i0 % 8));
343 bitmap->data[i1 / 8] &= ~(1 << (i1 % 8));
344
345 // Advance indices
346 v_idx = _mm_add_epi64(v_idx, v_step2);
347 idx += 2 * step;
348 }
349 }
350
351 // Handle remaining steps
352 for (; idx <= limit; idx += step)
353 {
354 bitmap->data[idx / 8] &= ~(1 << (idx % 8));
355 }
356
357#else
358 // Fallback to scalar implementation
359 for (uint64_t idx = start_idx; idx <= limit; idx += step)
360 {
361 bitmap->data[idx / 8] &= ~(1 << (idx % 8));
362 }
363#endif
364}

412int bitmap_validate_hash(BITMAP *bitmap)
413{
414 assert(bitmap && bitmap->data && "Invalid bitmap");
415
416 unsigned char correct_hash[SHA256_DIGEST_LENGTH]; // Buffer to hold the computed hash
417
418 // Generate SHA-256 hash and store it in correct_hash
419 SHA256((unsigned char *)bitmap->data, bitmap->byte_size, correct_hash);
420
421 // Compare actual_hash with the stored hash in bitmap->sha256
422 if (memcmp(correct_hash, bitmap->sha256, SHA256_DIGEST_LENGTH) == 0)
423 {
424 return 1; // SHA-256 match
425 }
426 else
427 {
428 log_error("SHA-256 checksum validation failed.");
429 return 0; // SHA-256 mismatch
430 }
431}

440int bitmap_fwrite(BITMAP *bitmap, FILE *file)
441{
442 assert(bitmap && bitmap->data && "Invalid bitmap passed to bitmap_fwrite.");
443 assert(file && "File pointer is NULL in bitmap_fwrite.");
444
445 // Write size
446 if (fwrite(&bitmap->size, sizeof(size_t), 1, file) != 1)
447 {
448 log_error("Failed to write bitmap size to file.");
449 return 0;
450 }
451
452 // Write data
453 if (fwrite(bitmap->data, sizeof(unsigned char), bitmap->byte_size, file) != bitmap->byte_size)
454 {
455 log_error("Failed to write bitmap data to file.");
456 return 0;
457 }
458
459 // Compute SHA-256 hash if not already computed
460 // Check if hash is all zeros by examining all bytes
461 int hash_is_zero = 1;
462 for (size_t i = 0; i < SHA256_DIGEST_LENGTH; i++)
463 {
464 if (bitmap->sha256[i] != 0)
465 {
466 hash_is_zero = 0;
467 break;
468 }
469 }
470
471 if (hash_is_zero)
472 {
473 bitmap_compute_hash(bitmap);
474 }
475
476 // Write SHA-256 hash
477 if (fwrite(bitmap->sha256, 1, SHA256_DIGEST_LENGTH, file) != SHA256_DIGEST_LENGTH)
478 {
479 log_error("Failed to write SHA-256 hash to file.");
480 return 0;
481 }
482
483 return 1; // Success
484}

493{

497 size_t size;

499 {

503

510

513 {
514 log_error("Failed to read complete bitmap data from file.");
515 bitmap_free(&bitmap);
516 return NULL;
517 }
518
519 // Read SHA-256 hash
520 if (fread(bitmap->sha256, 1, SHA256_DIGEST_LENGTH, file) != SHA256_DIGEST_LENGTH)
521 {
522 log_error("Failed to read SHA-256 hash from file.");
523 bitmap_free(&bitmap);
524 return NULL;
525 }
526

528 {
529 log_error("SHA-256 hash validation failed.");
530 bitmap_free(&bitmap);
