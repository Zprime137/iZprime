35uint64_t SiZ_stream(INPUT_SIEVE_RANGE *input_range)
36{
37 assert(input_range && input_range->start && "Invalid INPUT_SIEVE_RANGE passed to SiZ_stream.");
38
39 int has_output_file = (input_range->filepath && input_range->filepath[0] != '\0');
40 FILE *output = stdout; // default to stdout if no valid filepath is provided
41
42 if (has_output_file)
43 {
44 output = fopen(input_range->filepath, "w");
45 if (output == NULL)
46 {
47 log_error("Failed to open output file: %s", input_range->filepath);
48 return 0;
49 }
50 }
51
52 uint64_t total = 0; // output: total prime count
53
54 int vx = VX6; // Use VX6 segment size (1,616,615 bits) for optimal results
55 // Miller-Rabin rounds, bounded [5, 50]
56 int mr_rounds = MIN(MAX(input_range->mr_rounds, 5), 50);
57
58 IZM_RANGE_INFO info = range_info_init(input_range, vx);
59 if (info.y_range < 0)
60 {
61 if (has_output_file)
62 fclose(output);
63 range_info_free(&info);
64 return 0;
65 }
66
67 IZM *iZm = NULL;
68 mpz_t current_y;
69 mpz_init(current_y);
70 mpz_set(current_y, info.Ys);
71 int start_x = mpz_fdiv_ui(info.Xs, info.vx);
72 int end_x = mpz_fdiv_ui(info.Xe, info.vx);
73
74 // if Ys = 0, use SiZm for the first segment
75 if (mpz_cmp_ui(current_y, 0) == 0)
76 {
77 uint64_t limit = mpz_cmp_ui(info.Ye, 0) > 0 ? info.vx : end_x;
78 UI64_ARRAY *primes = SiZm(limit * 6 + 1);
79 if (!primes)
80 {
81 total = 0;
82 goto stream_cleanup;
83 }
84
85 uint64_t s = mpz_get_ui(info.Zs);
86 uint64_t e = mpz_get_ui(info.Ze);
87
88 for (int i = 0; i < primes->count; i++)
89 {
90 // only primes in [Zs, Ze]
91 if (primes->array[i] > s && primes->array[i] <= e)
92 {
93 total++;
94 fprintf(output, "%llu ", primes->array[i]);
95 }
96 }
97
98 start_x = 1; // next segment starts at x=1
99 mpz_add_ui(current_y, current_y, 1); // increment Ys for the next segment
100
101 ui64_free(&primes);
102 }
103
104 // No remaining vx segments after the initial one.
105 if (mpz_cmp(current_y, info.Ye) > 0)
106 {
107 goto stream_cleanup;
108 }
109
110 // Initialize iZm structure for vx segments
111 iZm = iZm_init(vx);
112 if (!iZm)
113 {
114 total = 0;
115 goto stream_cleanup;
116 }
117
118 // Process remaining segments for y in [current_y:Ye]
119 int first_segment = 1;
120 while (mpz_cmp(current_y, info.Ye) <= 0)
121 {
122 int seg_start_x = first_segment ? start_x : 1;
123 int seg_end_x = (mpz_cmp(current_y, info.Ye) == 0) ? end_x : vx;
124 char *y_str = mpz_get_str(NULL, 10, current_y);
125 if (!y_str)
126 {
127 total = 0;
128 goto stream_cleanup;
129 }
130
131 VX_SEG *vx_obj = vx_init(iZm, seg_start_x, seg_end_x, y_str, mr_rounds);
132 free(y_str);
133 if (!vx_obj)
134 {
135 // check logs for errors
136 total = 0;
137 goto stream_cleanup;
138 }
139
140 vx_stream(vx_obj, output);
141 total += vx_obj->p_count; // accumulate prime count
142
143 vx_free(&vx_obj);
144 first_segment = 0;
145 mpz_add_ui(current_y, current_y, 1); // increment Ys for the next segment
146 }
147
148stream_cleanup:
149 iZm_free(&iZm);
150 range_info_free(&info);
151 mpz_clear(current_y);
152 if (has_output_file)
153 fclose(output);
154 else
155 fflush(stdout);
156
157 return total;
158}

184uint64_t SiZ_count(INPUT_SIEVE_RANGE *input_range, int cores_num)
185{
186 assert(input_range && input_range->start && input_range->range > 100 &&
187 "Invalid INPUT_SIEVE_RANGE passed to SiZ_count.");
188
189 uint64_t total = 0;
190 int vx = compute_l2_vx(pow(10, 9)); // Use a segment width that balances workload and overhead for 10^9 range
191 cores_num = MIN(cores_num, get_cpu_cores_count());
192 IZM *iZm = NULL;
193 int (*pipe_fds)[2] = NULL;
194 pid_t *pids = NULL;
195
196 IZM_RANGE_INFO info = range_info_init(input_range, vx);
197 if (info.y_range < 0)
198 {
199 range_info_free(&info);
200 return 0;
201 }
202
203 mpz_t current_y;
204 mpz_init(current_y);
205 mpz_set(current_y, info.Ys);
206 int start_x = mpz_fdiv_ui(info.Xs, vx);
207 int end_x = mpz_fdiv_ui(info.Xe, vx);
208
209 // if current_y = 0, use sieve_iZm for the first segment
210 if (mpz_cmp_ui(current_y, 0) == 0)
211 {
212 uint64_t limit = mpz_cmp_ui(info.Ye, 0) > 0 ? vx : end_x;
213 UI64_ARRAY *primes = SiZm(limit * 6 + 1);
214 if (!primes)
215 {
216 total = 0;
217 goto count_cleanup;
218 }
219
220 total += primes->count;
221 uint64_t s = mpz_get_ui(info.Zs);
222 uint64_t e = mpz_get_ui(info.Ze);
223
224 for (int i = 0; i < primes->count; i++)
225 {
226 // Exclude values outside [Zs, Ze] from the first solid segment.
227 if (primes->array[i] < s || primes->array[i] > e)
228 {
229 total--;
230 }
231 }
232
233 ui64_free(&primes);
234 start_x = 1;
235 mpz_add_ui(current_y, current_y, 1); // increment Ys for the next segment
236 }
237
238 if (mpz_cmp(current_y, info.Ye) > 0)
239 {
240 goto count_cleanup;
241 }
242
243 iZm = iZm_init(vx);
244 if (!iZm)
245 {
246 total = 0;
247 goto count_cleanup;
248 }
249
250 // Handle edge cases:
251 mpz_t prime_z;
252 mpz_init(prime_z);
253 // if Ys > 0 and Zs % 6 <= 1
254 if (mpz_cmp_ui(current_y, 0) > 0 && mpz_fdiv_ui(info.Zs, 6) <= 1)
255 {
256 // if iZ(Xs, -1) < Zs and prime, decrement total
257 iZ_mpz(prime_z, info.Xs, -1);
258 if (mpz_cmp(prime_z, info.Zs) < 0)
259 {
260 if (mpz_probab_prime_p(prime_z, 25))
261 {
262 total--;
263 }
264 }
265 }
266 // if Ye > 0 and Ze % 6 <= 1
267 if (mpz_cmp_ui(info.Ye, 0) > 0 && mpz_fdiv_ui(info.Ze, 6) <= 1)
268 {
269 // if iZ(Xe, 1) > Ze and prime, decrement total
270 iZ_mpz(prime_z, info.Xe, 1);
271 if (mpz_cmp(prime_z, info.Ze) > 0)
272 {
273 if (mpz_probab_prime_p(prime_z, 25))
274 {
275 total--;
276 }
277 }
278 }
279 mpz_clear(prime_z);
280
281 int total_segments = (int)(mpz_get_ui(info.Ye) - mpz_get_ui(current_y) + 1);
282 if (total_segments <= 0)
283 {
284 goto count_cleanup;
285 }
286
287 // Single-process processing of all segments
288 if (cores_num == 1)
289 {
290 int first_segment = 1;
291 for (int i = 0; i < total_segments; i++)
292 {
293 int seg_start_x = first_segment ? start_x : 1;
294 int seg_end_x = (i == total_segments - 1) ? end_x : vx;
295 char *y_str = mpz_get_str(NULL, 10, current_y);
296 if (!y_str)
297 {
298 total = 0;
299 goto count_cleanup;
300 }
301
302 VX_SEG *vx_obj = vx_init(iZm, seg_start_x, seg_end_x, y_str, input_range->mr_rounds);
303 free(y_str);
304 if (!vx_obj)
305 {
306 total = 0;
307 goto count_cleanup;
308 }
309
310 vx_full_sieve(vx_obj, 0);
311 total += vx_obj->p_count;
312
313 vx_free(&vx_obj);
314 first_segment = 0;
315 mpz_add_ui(current_y, current_y, 1); // increment Ys for the next segment
316 }
317 goto count_cleanup;
318 }
319
320 // Multi-process processing of remaining segments
321 // if total_segments < cores_num, adjust cores_num
322 if (total_segments < cores_num)
323 {
324 cores_num = total_segments;
325 }
326
327 int segments_per_core = total_segments / cores_num;
328 int remainder_segments = total_segments % cores_num;
329
330 pipe_fds = malloc((size_t)cores_num * sizeof(*pipe_fds));
331 pids = malloc((size_t)cores_num * sizeof(*pids));
332 if (!pipe_fds || !pids)
333 {
334 log_error("SiZ_count: Failed to allocate process bookkeeping arrays.");
335 total = 0;
336 goto count_cleanup;
337 }
338
339 for (int i = 0; i < cores_num; i++)
340 {
341 pids[i] = -1;
342 pipe_fds[i][0] = -1;
343 pipe_fds[i][1] = -1;
344 }
345
346 for (int core = 0; core < cores_num; core++)
347 {
348 if (pipe(pipe_fds[core]) == -1)
349 {
350 log_error("SiZ_count: Failed to create pipe for core %d. Aborting.", core);
351 // cleanup and abort on any pipe error
352 for (int j = 0; j <= core; j++)
353 {
354 if (pipe_fds[j][0] != -1)
355 close(pipe_fds[j][0]);
356 if (pipe_fds[j][1] != -1)
357 close(pipe_fds[j][1]);
358 }
359 total = 0;
360 goto count_cleanup;
361 }
362
363 pids[core] = fork();
364 if (pids[core] < 0)
365 {
366 log_error("SiZ_count: Failed to fork process for core %d. Aborting.", core);
367 // close this core's pipe and any previous ones, then exit
368 for (int j = 0; j <= core; j++)
369 {
370 if (pipe_fds[j][0] != -1)
371 close(pipe_fds[j][0]);
372 if (pipe_fds[j][1] != -1)
373 close(pipe_fds[j][1]);
374 }
375 total = 0;
376 goto count_cleanup;
377 }
378
379 if (pids[core] == 0)
380 {
381 // Child process
382 // Close unrelated pipes
383 for (int j = 0; j < cores_num; j++)
384 {
385 if (j != core && pipe_fds[j][0] != -1)
386 close(pipe_fds[j][0]);
387 if (j != core && pipe_fds[j][1] != -1)
388 close(pipe_fds[j][1]);
389 }
390 close(pipe_fds[core][0]); // close read end
391
392 // Compute this child's starting Y and number of segments
393 int offset = core * segments_per_core + (core < remainder_segments ? core : remainder_segments);
394 int local_segments = segments_per_core + (core < remainder_segments ? 1 : 0);
395
396 uint64_t child_total = 0;
397
398 if (local_segments > 0)
399 {
400 mpz_t local_Ys;
401 mpz_init(local_Ys);
402 mpz_set(local_Ys, current_y);
403 mpz_add_ui(local_Ys, local_Ys, offset);
404
405 // Each child has its own IZM to avoid data races
406 IZM *iZm_local = iZm_clone(iZm);
407 if (!iZm_local)
408 {
409 mpz_clear(local_Ys);
410 close(pipe_fds[core][1]);
411 exit(1);
412 }
413
414 for (int i = 0; i < local_segments; i++)
415 {
416 int global_segment = offset + i;
417 int seg_start_x = (global_segment == 0) ? start_x : 1;
418 int seg_end_x = (global_segment == total_segments - 1) ? end_x : vx;
419 char *y_str = mpz_get_str(NULL, 10, local_Ys);
420 if (!y_str)
421 {
422 iZm_free(&iZm_local);
423 mpz_clear(local_Ys);
424 close(pipe_fds[core][1]);
425 exit(1);
426 }
427
428 VX_SEG *vx_obj = vx_init(iZm_local, seg_start_x, seg_end_x, y_str, input_range->mr_rounds);
429 free(y_str);
430 if (!vx_obj)
431 {
432 iZm_free(&iZm_local);
433 mpz_clear(local_Ys);
434 close(pipe_fds[core][1]);
435 exit(1);
436 }
437
438 vx_full_sieve(vx_obj, 0);
439 child_total += vx_obj->p_count;
440
441 vx_free(&vx_obj);
442 mpz_add_ui(local_Ys, local_Ys, 1);
443 }
444
445 iZm_free(&iZm_local);
446 mpz_clear(local_Ys);
447 }
448
449 // Send result to parent
450 ssize_t written = write(pipe_fds[core][1], &child_total, sizeof(child_total));
451 if (written != sizeof(child_total))
452 {
453 log_error("SiZ_count: Child %d failed to write result.", core);
454 }
455
456 close(pipe_fds[core][1]);
457 exit(0);
458 }
459 else
460 {
461 // Parent process
462 close(pipe_fds[core][1]); // close write end
463 pipe_fds[core][1] = -1;
464 }
465 }
466
467 // Collect results
468 for (int core = 0; core < cores_num; core++)
469 {
470 if (pids[core] > 0 && pipe_fds[core][0] != -1)
471 {
472 uint64_t child_total = 0;
473 ssize_t bytes_read = read(pipe_fds[core][0], &child_total, sizeof(child_total));
474 if (bytes_read == sizeof(child_total))
475 {
476 total += child_total;
477 }
478 else if (bytes_read == 0)
479 {
480 log_error("SiZ_count: Child %d closed pipe without sending result.", core);
481 }
482 else if (bytes_read < 0)
483 {
484 log_error("SiZ_count: Failed to read result from child %d.", core);
485 }
486 else
487 {
488 log_error("SiZ_count: Partial read from child %d (got %zd bytes).", core, bytes_read);
489 }
490
491 close(pipe_fds[core][0]);
492 pipe_fds[core][0] = -1;
493
494 int status;
495 waitpid(pids[core], &status, 0);
496 if (WIFEXITED(status) && WEXITSTATUS(status) != 0)
497 {
498 log_error("SiZ_count: Child %d exited with status %d.", core, WEXITSTATUS(status));
499 // propagate child failure as an overall error
500 total = 0;
501 goto count_cleanup;
502 }
503 else if (WIFSIGNALED(status))
504 {
505 log_error("SiZ_count: Child %d terminated by signal %d.", core, WTERMSIG(status));
506 total = 0;
507 goto count_cleanup;
508 }
509 }
510 else if (pipe_fds[core][0] != -1)
511 {
512 close(pipe_fds[core][0]);
513 pipe_fds[core][0] = -1;
514 }
515 }
516
517count_cleanup:
518 free(pids);
519 free(pipe_fds);
520 range_info_free(&info);
521 iZm_free(&iZm);
522 mpz_clear(current_y);
523
524 return total;
525}

545int vy_random_prime(mpz_t p, int bit_size, int cores_num)
546{
547 int found = 0; // flag to indicate if a prime was found
548 bit_size = MAX(bit_size, 10); // Set minimum bit size
549
550 // 1. Compute max vx for the given bit-size
551 mpz_t vx;
552 mpz_init(vx);
553 compute_max_vx(vx, bit_size);
554
555 // 2. If < 2 cores, run the search in-process
556 if (cores_num < 2)
557 {
558 found = vy_search_prime(p, 0, vx);
559 mpz_clear(vx);
560 return found;
561 }
562
563 // 3. Else, fork multiple processes to search for a prime
564 // Create a pipe for inter-process communication.
565 int fd[2];
566 if (pipe(fd) == -1)
567 {
568 log_error("Failed to create pipe in vy_random_prime. Falling back to in-process search.");
569 found = vy_search_prime(p, 0, vx);
570 mpz_clear(vx);
571 return found;
572 }
573
574 pid_t pids[cores_num];
575 int forks_created = 0;
576
577 // Fork child processes.
578 for (int i = 0; i < cores_num; i++)
579 {
580 pid_t pid = fork();
581 if (pid < 0)
582 {
583 log_error("Failed to fork process %d in vy_random_prime", i);
584 continue; // Skip this fork failure and continue creating other forks
585 }
586 else if (pid == 0)
587 {
588 // Child process: close the read-end.
589 close(fd[0]);
590
591 mpz_t local_p; // local prime candidate
592 mpz_init(local_p);
593
594 // Search for a candidate prime
595 found = vy_search_prime(local_p, 0, vx);
596
597 // If a candidate is found, send it via the pipe.
598 if (found)
599 {
600 char *p_str = mpz_get_str(NULL, 10, local_p);
601 if (p_str)
602 {
603 write(fd[1], p_str, strlen(p_str) + 1);
604 free(p_str);
605 }
606 }
607 mpz_clear(local_p);
608
609 close(fd[1]); // Close the write-end.
610 exit(0);
611 }
612 else
613 {
614 // Parent process saves child's PID.
615 pids[i] = pid;
616 forks_created++;
617 }
618 }
619
620 // 4. Parent reads first result from the pipe.
621 close(fd[1]); // Close the write-end.
622
623 if (forks_created == 0)
624 {
625 log_error("No child processes were created in vy_random_prime, falling back to in-process search");
626 close(fd[0]);
627 found = vy_search_prime(p, 0, vx);
628 mpz_clear(vx);
629 return found;
630 }
631
632 // Allocate a sufficiently large buffer to hold the prime string + null terminator
633 // bit_size * 0.302 is approx digits, add extra padding for safety
634 size_t buf_size = (size_t)(bit_size * 0.4);
635 char *buf = malloc(buf_size);
636 if (!buf)
637 {
638 log_error("Failed to allocate buffer in vy_random_prime");
639 found = 0;
640 }
641 else
642 {
643 ssize_t n = read(fd[0], buf, buf_size - 1);
644 if (n == -1)
645 {
646 log_error("Failed to read from pipe in vx_random_prime");
647 found = 0;
648 }
649 else if (n > 0)
650 {
651 buf[n] = '\0'; // Ensure null-termination
652 if (mpz_set_str(p, buf, 10) == 0)
653 {
654 found = 1;
655 }
656 else
657 {
658 log_error("Failed to set prime from buffer in vx_random_prime");
659 found = 0;
660 }
661 }
662 free(buf);
663 }
664 close(fd[0]); // Close the read-end of the pipe.
665
666 // 5. Terminate all child processes
667 for (int i = 0; i < forks_created; i++)
668 {
669 kill(pids[i], SIGTERM); // Terminate child process
670 waitpid(pids[i], NULL, 0); // Wait for child process to terminate
671 }
672
673 mpz_clear(vx);
674
675 return found;
676}

692int vx_random_prime(mpz_t p, int bit_size, int cores_num)
693{

700 {

802 waitpid(pids[i], NULL, 0);

804
805 return found;

819int iZ_next_prime(mpz_t p, mpz_t base, int forward)
820{
821 // 1. Initialization
822 int found = 0; // flag to indicate if a prime was found
823
824 // tmp variable to hold the value until a prime is found
825 mpz_t z;
826 mpz_init_set(z, base); // set tmp = base
827
828 // a. Edge cases:
829 // if forward and base is iZ-, check next iZ+
830 if (mpz_fdiv_ui(z, 6) == 5 && forward)
831 {
832 mpz_add_ui(z, z, 2); // increment tmp by 2
833 if (mpz_probab_prime_p(z, MR_ROUNDS))
834 {
835 mpz_set(p, z); // set p = tmp + 2
836 mpz_clear(z);
837 return 1;
838 }
839 }
840 // if backward and base is iZ+, check previous iZ-
841 else if (mpz_fdiv_ui(z, 6) == 1 && !forward)
842 {
843 mpz_sub_ui(z, z, 2); // decrement tmp by 2
844 if (mpz_probab_prime_p(z, MR_ROUNDS))
845 {
846 mpz_set(p, z); // set p = tmp - 2
847 mpz_clear(z);
848 return 1;
849 }
850 }
851
852 int vx = (mpz_sizeinbase(base, 2) > 2048) ? VX6 : VX5;
853 IZM *iZm = iZm_init(vx);
854 if (!iZm)
855 {
856 log_error("iZm initialization failed in iZ_next_prime.");
857 mpz_clear(z);
858 return 0;
859 }
860
861 // c. Initialize and set y and yvx
862 mpz_t y, yvx, x_p;
863 mpz_init(y);
864 mpz_init(yvx);
865 mpz_init(x_p);
866
867 mpz_div_ui(y, base, 6 * vx); // compute y = base / 6 * vx
868 mpz_mul_ui(yvx, y, vx); // compute yvx = y * vx
869 mpz_div_ui(x_p, z, 6); // compute x_p = tmp / 6
870
871 // 2. Iterate over the x5 and x7 bitmaps to find a prime
872 // set start_x = x_p % vx +/- 1
873 int step = forward ? 1 : -1;
874 int start_x = mpz_fdiv_ui(x_p, vx) + step;
875 int end_x = forward ? vx : 1;
876
877 int i = 0; // segment counter
878
879 while (!found)
880 {
881 if (forward)
882 {
883 if (i > 0)
884 start_x = 1; // start from the beginning of the bitmap
885
886 for (int x = start_x; x <= end_x; x++)
887 {
888 // check if x5[x] is set
889 if (bitmap_get_bit(iZm->base_x5, x))
890 {
891 mpz_add_ui(x_p, yvx, x); // set x_p = yvx + x
892 iZ_mpz(z, x_p, -1); // compute p = iZ(x_p, -1)
893 // check if tmp is prime
894 found = mpz_probab_prime_p(z, MR_ROUNDS);
895
896 if (found)
897 break;
898 }
899
900 // check if x7[x] is set
901 if (bitmap_get_bit(iZm->base_x7, x))
902 {
903 mpz_add_ui(x_p, yvx, x); // set x_p = yvx + x
904 iZ_mpz(z, x_p, 1); // compute tmp = iZ(x_p, 1)
905 // check if tmp is prime
906 found = mpz_probab_prime_p(z, MR_ROUNDS);
907
908 if (found)
909 break;
910 }
911 }
912
913 mpz_add_ui(yvx, yvx, vx); // increment yvx by vx for next segment
914 }
915 else // backward search
916 {
917 if (i > 0)
918 start_x = vx; // start from the end of the bitmaps
919
920 // check iZ+ first if backward
921 for (int x = start_x; x >= end_x; x--)
922 {
923 // check if x7[x] is set
924 if (bitmap_get_bit(iZm->base_x7, x))
925 {
926 mpz_add_ui(x_p, yvx, x); // set x_p = yvx + x
927 iZ_mpz(z, x_p, 1); // compute tmp = iZ(x_p, 1)
928 // check if tmp is prime
929 found = mpz_probab_prime_p(z, MR_ROUNDS);
930
931 if (found)
932 break;
933 }
934
935 // check iZ-
936 if (bitmap_get_bit(iZm->base_x5, x))
937 {
938 mpz_add_ui(x_p, yvx, x); // set x_p = yvx + x
939 iZ_mpz(z, x_p, -1); // compute p = iZ(x_p, -1)
940 // check if tmp is prime
941 found = mpz_probab_prime_p(z, MR_ROUNDS);
942
943 if (found)
944 break;
945 }
946 }
947
948 mpz_sub_ui(yvx, yvx, vx); // decrement yvx by vx for next segment
949 }
950
951 i++; // increment segment counter
952 }
953
954 // 3. Set the found prime
955 if (found)
956 mpz_set(p, z); // set p = tmp
957 else
958 log_debug("No prime found :/");
959
960 // cleanup
961 iZm_free(&iZm);
962 mpz_clears(y, yvx, x_p, z, NULL);
963
964 return found;
965}
