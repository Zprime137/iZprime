72void log_rotate(const char *log_file, size_t max_size)
73{
74 struct stat st;
75 if (stat(log_file, &st) == 0 && st.st_size >= (off_t)max_size) // Cast max_size to off_t to match types
76 {
77 char old_log_file[256];
78 for (int i = 5; i > 0; --i)
79 {
80 snprintf(old_log_file, sizeof(old_log_file), "%s.%d", log_file, i);
81 if (i == 5)
82 {
83 remove(old_log_file); // Remove the oldest log file
84 }
85 else
86 {
87 char prev_log_file[256];
88 snprintf(prev_log_file, sizeof(prev_log_file), "%s.%d", log_file, i + 1);

145static void log_message_va(LogLevel level, const char *format, va_list args)
146{
147 if (level < current_log_level)
148 return; // Don't log messages below the current log level
149
150 char message[1024];
151 // Format string comes from a variadic function parameter - suppress warning
152#pragma clang diagnostic push
153#pragma clang diagnostic ignored "-Wformat-nonliteral"
154 vsnprintf(message, sizeof(message), format, args);
155#pragma clang diagnostic pop
156
157 char timestamp[20];
158 get_current_timestamp(timestamp, sizeof(timestamp));
159
160 pthread_mutex_lock(&log_mutex);
161
162 FILE *file = fopen(LOG_FILE, "a");
163 if (!file)
164 {
165 perror("Failed to open log file");
166 pthread_mutex_unlock(&log_mutex); // Ensure mutex is unlocked
167 return;
168 }
169
170 fprintf(file, "[%s] [%s] %s\n", timestamp, log_level_to_string(level), message);
171 fclose(file);
172
173 pthread_mutex_unlock(&log_mutex);
174}
