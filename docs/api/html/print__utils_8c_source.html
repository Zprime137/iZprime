30void print_centered_text(const char *text, int line_length, char fill_char)
31{
32 if (text == NULL)
33 text = "";
34
35 int text_length = (int)strlen(text);
36 if (text_length >= line_length)
37 {
38 printf("%s\n", text);
39 return;
40 }
41
42 int left_padding = (line_length - text_length) / 2;
43 int right_padding = line_length - text_length - left_padding;
44
45 for (int i = 0; i < left_padding; ++i)
46 printf("%c", fill_char);
47 printf("%s", text);
48 for (int i = 0; i < right_padding; ++i)
